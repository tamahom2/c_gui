153  ei_event_set_active_widget(ei_app_root_widget());
154  while(!quit) {
155  struct ei_event_t *event = malloc(sizeof(struct ei_event_t));
156  //First we lock the surface

159  //We should add pick_surface after
160  //visit all widgets
161  ei_widget_t visiting = ROOT_WIDGET;
162  ei_impl_widget_draw_children(visiting,ROOT_SURFACE,OFFSCREEN_PICKING,&CLIPPER_TOTAL);
163 
164  //We unlock the surface and update
165  hw_surface_unlock(OFFSCREEN_PICKING);
166  hw_surface_unlock(ROOT_SURFACE);
167  if(RECT_UPDATE!=NULL) {
168  hw_surface_update_rects(ROOT_SURFACE, NULL);
169  }
170  RECT_UPDATE = NULL;
171  visiting = ROOT_WIDGET;
172  //then we wait for events
173  hw_event_wait_next(event);
174  //If it's a mouse click we should change the active widget
175  if(event->type==ei_ev_mouse_buttondown){
176  ei_point_t mouse_click = event->param.mouse.where;
177  ei_event_set_active_widget(ei_widget_pick(&mouse_click));
178  }
179  visiting = ei_event_get_active_widget();
180  ei_default_handle_func_t default_handler_func = ei_event_get_default_handle_func();
181  if(visiting!=NULL && visiting->wclass->handlefunc==NULL && default_handler_func!=NULL) default_handler_func(event);
182  else if(visiting!=NULL && visiting->wclass->handlefunc != NULL){
183  if(!visiting->wclass->handlefunc(visiting,event)) default_handler_func(event);
184  }else{
185  if(default_handler_func != NULL) default_handler_func(event);
186  }
