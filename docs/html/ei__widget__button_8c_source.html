210 bool ei_button_handlefunc(ei_widget_t widget,struct ei_event_t* event){
211  if(event->type ==ei_ev_mouse_buttondown){

217  return true;
218  }
219  if(event->type == ei_ev_mouse_move){
220  if(((ei_button_t *) widget)->callback != NULL){

222  }
223  ei_app_invalidate_rect(&widget->screen_location);
224  return true;
225  }

227  ((ei_button_t *) widget)->relief = ei_relief_raised;
228  ei_app_invalidate_rect(&widget->screen_location);
229  ei_event_set_active_widget(ei_app_root_widget());
230  return true;
231  }
232  return false;
