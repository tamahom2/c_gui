53 void resize_toplevel(ei_widget_t widget, ei_event_t* event, ei_user_param_t user_param)
54 {
55  if(event->type == ei_ev_mouse_buttondown) {
56  OLD_POSITION = event->param.mouse.where;
57  }
58  if(event->type == ei_ev_mouse_move){
59  ei_point_t new_position = event->param.mouse.where;
60  int new_width = widget->parent->screen_location.size.width;
61  int new_height = widget->parent->screen_location.size.height;
62  if(((ei_toplevel_t *) widget->parent)->resizable == ei_axis_both || ((ei_toplevel_t *) widget->parent)->resizable == ei_axis_x) {
63  new_width += new_position.x - OLD_POSITION.x;
64  }
65  if(((ei_toplevel_t *) widget->parent)->resizable == ei_axis_both || ((ei_toplevel_t *) widget->parent)->resizable == ei_axis_y) {
66  new_height += new_position.y - OLD_POSITION.y;
67  }
68  ei_place(widget->parent,NULL,NULL,NULL, &(int){new_width},&(int){new_height},NULL,NULL,NULL,NULL);
69  OLD_POSITION = new_position;
70  }
71  if(event->type == ei_ev_mouse_buttonup){
72  ei_event_set_active_widget(widget->parent->parent);
73  }
74 }

239 bool ei_toplevel_handlefunc(ei_widget_t widget,struct ei_event_t* event){
240  if (event->type == ei_ev_mouse_buttondown) {
241  OLD_POSITION = event->param.mouse.where;
242  return true;
243  }
244  else if (event->type == ei_ev_mouse_move) {
245  ei_point_t new_position = event->param.mouse.where;
246  int x_deviation = new_position.x - OLD_POSITION.x;
247  int y_deviation = new_position.y - OLD_POSITION.y;
248  ei_place_xy(widget, widget->screen_location.top_left.x + x_deviation,
249  widget->screen_location.top_left.y + y_deviation);
250  OLD_POSITION = new_position;
251  }
252  else if (event->type == ei_ev_mouse_buttonup) {
253  ei_event_set_active_widget(widget->parent);
254  }
255  return false;
256 }
